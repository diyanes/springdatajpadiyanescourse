SELECT
    BILLING_ACCOUNT_ID, PAYMENT_RECEIVER, BILLING_ACCOUNT_NUMBER,SUBSCRIBER_ID, 
CASE WHEN AUTO_DRAFT.MEMBER_BILLING_GRANT_STATUS_INDICATOR= 'Y' THEN 'Y' ELSE 'N' END AS "AUTO_PAY_ENABLED",
PAYMENT_METHOD.PAYMENT_METHOD_ID ,PAYMENT_METHOD.PAYMENT_METHOD_TYPE ,EFFECTIVE_DATE 
FROM
    MEMBER_BILLING_ACCOUNT
LEFT JOIN PAYMENT_SETUP ON
    PAYMENT_SETUP.MEMBER_BILLING_ACCOUNT_ID = MEMBER_BILLING_ACCOUNT.BILLING_ACCOUNT_ID
LEFT JOIN PAYMENT_METHOD ON
    PAYMENT_METHOD.PAYMENT_METHOD_ID= PAYMENT_SETUP.PAYMENT_METHOD_ID
LEFT JOIN AUTO_DRAFT ON
    AUTO_DRAFT.PAYMENT_SETUP_ID =  PAYMENT_SETUP.PAYMENT_SETUP_ID
WHERE PAYMENT_SETUP.IS_DEFAULT = 'Y' 
AND MEMBER_BILLING_ACCOUNT.SUBSCRIBER_ID = '90002'
--AND AUTO_DRAFT.MEMBER_BILLING_GRANT_STATUS_INDICATOR= 'Y'
ORDER BY
BILLING_ACCOUNT_ID ASC;
